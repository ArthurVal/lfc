  $<BUILD_INTERFACE:${CMAKE_BINARY_DIR}/include>
  $<INSTALL_INTERFACE:${CMAKE_INSTALL_INCLUDEDIR}>

  # PRIVATE

)

  -Wpedantic
  -Wshadow
  -Wdouble-promotion
  -Wundef

  # -Werror
)
